<critical>
**VERIFIED build-status enum (D-07, 02-RESEARCH.md:61-92).** The v1 build resource serializes the
PUBLIC enum only: `queued | building | ready | failed`. Terminal-SUCCESS = `ready`. Terminal-
FAILURE = `failed`. Do NOT match `succeeded`, `success`, `done`, `complete`, `in_review`, or
`rejected` — none appear on the v1 build resource. Anything that is not `ready`/`failed` => keep
polling.

**Mapper coarsening pitfall (02-RESEARCH.md:92, Pitfall 2):** the internal terminal `rejected`
state coarsens to public `building` (the mapper's default branch). A build rejected internally
polls as `building` forever — only `--timeout` catches it. The timeout message MUST point the
user at `appo status <id>` (the app overview surfaces publication_state/rejection) in addition to
`appo status <id> --build <buildId>`.

**Where "missing Apple credential" actually blocks (02-RESEARCH.md:111-140):** NOT at publish.
It blocks at the BUILD-TRIGGER step (`triggerBuild` -> 422 prerequisite_failed with codes like
APPLE_CREDENTIALS_MISSING / APPLE_ENROLLMENT_INCOMPLETE / ICONS_MISSING / APP_BLOCKED /
SUBSCRIPTION_INACTIVE / APPLE_TEAM_MISSING / ASC_APP_ID_MISSING, each carrying
details.next_action + details.dashboard_url) BEFORE any build is created. Route it through
`renderError`. Publish itself only emits 204 / 409 conflict / 422 APP_BLOCKED / 404.

**Confirm-gate is high-severity — never weaken it (02-RESEARCH.md Pitfall 5, security_context):**
no `--yes`/`--confirm` => publish step is `gated`, NO publish POST issued, exit 3. `--json` only
changes output; it NEVER auto-confirms. `--yes`/`--confirm` is the ONLY path to confirm:true for
the publish op.

**Do NOT call `confirmGate` inside ship — reimplement the gate DECISION (key reuse note).**
`confirmGate` keys ONLY off `flags.confirm` (it would ignore `--yes`) AND it emits its own
competing `--json` object — both break ship's contract (ship accepts `--yes`/`--confirm` and emits
a SINGLE `{steps[], final_state}` ledger). Ship therefore REUSES `printPreview` verbatim but
reimplements the gate decision: `const wantYes = flags.yes === true || flags.confirm === true;`
and, when `!wantYes`, calls `printPreview` (human mode), records `gated`, and returns exit 3 with
NO publish POST. The invariant — no publish POST without `--yes`/`--confirm`, exit 3 — is what the
test asserts (no `confirmGate` call).

1. Poll loop (THE one new mechanism — 02-RESEARCH.md Pattern 2, terminal strings VERIFIED):
```js
const realSleep = (ms) => new Promise((r) => setTimeout(r, ms));

/** Poll a build to terminal. Public build-status enum: queued|building|ready|failed
 *  (VERIFIED — 02-RESEARCH.md). ready == terminal success, failed == terminal failure;
 *  anything else keeps polling. sleep/intervalMs/timeoutMs are injectable so tests run
 *  instantly. onChange streams a line only on status change (D-06). */
export async function pollBuild(apiBase, appId, buildId, {
  intervalMs = 5000, timeoutMs = 1_800_000, sleep = realSleep, onChange = () => {},
} = {}) {
  const start = Date.now();
  let last = null;
  for (;;) {
    const build = await ops.getBuild(apiBase, appId, buildId);
    const status = build?.status;
    if (status !== last) { onChange(status, build); last = status; }
    if (status === 'ready')  return { outcome: 'ready', build };
    if (status === 'failed') return { outcome: 'failed', build };
    if (Date.now() - start >= timeoutMs) return { outcome: 'timeout', build, last_status: status };
    await sleep(intervalMs);
  }
}
```
Note the timeout check is placed AFTER the terminal checks and BEFORE the sleep, so `timeoutMs:0`
with a single non-terminal `building` response returns `timeout` after one poll (the test path).

3. Ledger + EXIT map (02-RESEARCH.md Pattern 3 — one ledger drives both human stream and --json):
```js
/** One ledger drives both the human stream and the --json summary (D-11/D-12).
 *  In human mode each log() prints live (ASCII markers only — `->`, no unicode).
 *  In --json mode the stream is suppressed; the whole ledger + final_state is
 *  emitted once at completion. */
function shipReport(json) {
  const steps = [];
  const log = (line) => { if (!json) console.log(line); };
  const record = (step) => { steps.push(step); };
  const finish = (final_state, exitCode) => {
    if (json) console.log(JSON.stringify({ steps, final_state }));
    return exitCode;
  };
  return { log, record, finish };
}
```

```js
  const hasId = sub && !sub.startsWith('--');
  if (!hasId && (!flags.url || !flags.name)) {
    console.error('Usage: appo ship <id> | appo ship --url <u> --name <n> [--stores <list>] [--platform ios|android|all] [--yes] [--timeout <s>] [--json]');
    return 2;  // D-13 usage error — BEFORE any HTTP and BEFORE any ledger (see note below)

  const json = flags.json === true;
  const wantYes = flags.yes === true || flags.confirm === true;  // reimplements the gate decision (confirmGate checks only flags.confirm + emits its own --json — do NOT call it)
  const stores = parseStores(flags.stores);
  const { log, record, finish } = shipReport(json);

  // Helper: in --json mode, a thrown prerequisite_failed/conflict is caught locally so
  // ONE ledger object still emits (D-12 / A4). In human mode, rethrow to the top-level
  // try/catch -> renderError (Blocked/Next lines).
  const handleBlock = (err, step) => {
    if (!json) throw err;  // -> run() catch -> renderError -> 1
    record({ step, status: 'blocked', code: err.envelope?.code, message: err.message });
    return finish('blocked', 1);
  };

  let appId = hasId ? sub : null;

  // STEP create (new-app form only). UrlReachable 422 throws -> block (no app exists yet).

      app = await ops.createApp(apiBase, {
        name: flags.name, base_url: flags.url,
        metadata_name: flags['meta-name'], metadata_description: flags['meta-desc'],
      });

    appId = app.id;
    record({ step: 'create', status: 'ok', app_id: appId });
    log(`> create ... ok app #${appId}`);

    return handleBlock(err, 'build');

  record({ step: 'build', status: 'ok', build_id: buildId });
  log(`> build #${buildId} ... ${build.status}`);

  // STEP poll
  const res = await pollBuild(apiBase, appId, buildId, {
    timeoutMs: (Number(flags.timeout) || 1800) * 1000,
    onChange: (s) => log(`  ${s} -> ...`),
  });
  if (res.outcome === 'failed') {
    record({ step: 'build', status: 'failed', build_id: buildId });
    log(`x build failed. Next: appo fix-recipe ${appId}  (or: appo rejection ${appId})`);
    return finish('failed', 1);
  }
  if (res.outcome === 'timeout') {
    record({ step: 'build', status: 'timeout', build_id: buildId, last_status: res.last_status });

    return finish('failed', 1);
```
