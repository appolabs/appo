```javascript
import { test, expect } from 'vitest';
import { execSync } from 'node:child_process';
import { readFileSync } from 'node:fs';

test('npm pack ships only the whitelisted runtime files', () => {
  const out = JSON.parse(execSync('npm pack --dry-run --json', { encoding: 'utf-8' }));
  const paths = out[0].files.map((f) => f.path);
  for (const p of ['bin/appo.mjs','src/api.mjs','src/cli.mjs','src/config.mjs','src/login.mjs','src/ops.mjs','README.md','package.json']) {
    expect(paths).toContain(p);
  }
  for (const banned of [/^test\//, /\.planning/, /package-lock\.json/, /\.eslintrc/, /tsconfig\.json/, /vitest\.config/]) {
    expect(paths.some((p) => banned.test(p))).toBe(false);
  }
});

test('package.json carries the publish metadata + build-free quality gate + no runtime deps', () => {
  const p = JSON.parse(readFileSync('package.json', 'utf-8'));
  expect(p.publishConfig.access).toBe('public');
  expect(p.repository && p.homepage && p.bugs && p.author).toBeTruthy();
  expect(Array.isArray(p.keywords) && p.keywords.length).toBeTruthy();
  expect(p.scripts.prepublishOnly).toBe('npm run lint && npm run typecheck && npm test');
  expect(p.scripts.prepublishOnly).not.toMatch(/build/);
  expect(p.files).toContain('llms.txt');
  expect(p.dependencies && Object.keys(p.dependencies).length || 0).toBe(0);
});

test('release.yml mirrors the SDK trusted-publishing flow (npm, no build)', () => {
  const yml = readFileSync('.github/workflows/release.yml', 'utf-8');
  expect(yml).toMatch(/id-token: write/);
  expect(yml).toMatch(/npm publish --provenance --access public/);
  expect(yml).toMatch(/npm ci/);
  expect(yml).not.toMatch(/pnpm/i);
  expect(yml).not.toMatch(/run: .*build/);
});
```
NOTE: do NOT assert a strict total-file count or llms.txt presence here — llms.txt arrives in Plan 03.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions → npm registry | release.yml publishes via OIDC (no long-lived secret) |
| local dev → tarball | `npm pack` selects which files become public via the `files` whitelist |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-05 | Spoofing/Tampering | release.yml publish | mitigate | OIDC trusted publishing (`permissions: id-token: write`) + `--provenance` — NO long-lived `NPM_TOKEN` secret to leak or rotate; provenance attests the build origin. The one-time trusted-publisher registration is the user's documented action (D-09). |
| T-06-06 | Information Disclosure | npm pack tarball | mitigate | The `files` whitelist ships ONLY bin/src/README.md/llms.txt/package.json; the packaging test asserts no test/.planning/lockfile/config leaks into the published tarball (no source-of-secrets, no internal planning notes). |
| T-06-07 | Elevation (irreversible release) | executor running publish | mitigate | D-09: the executor stops at `npm pack --dry-run`; no `npm publish`, no tag push, no trusted-publisher registration. The packaging test uses `--dry-run --json` only. The live release is the user's explicit action. |
| T-06-08 | Tampering | prepublishOnly gate | accept | The gate (lint+typecheck+test) runs at real publish time only; it cannot block the executor's dry-run (npm pack does not run prepublishOnly — Pitfall 6). Acceptable: CI re-runs the same gate. |
</threat_model>
